#include <iostream>
using namespace std;

int main() {
    int num1, num2, num3, num4, num5, sum;

    // Input 5 numbers
    cout << "Enter five numbers: ";
    cin >> num1 >> num2 >> num3 >> num4 >> num5;

    // Calculate the sum
    sum = num1 + num2 + num3 + num4 + num5;

    // Display the result
    cout << "The sum of the numbers is: " << sum << endl;

    return 0;
}
